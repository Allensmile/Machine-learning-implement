$p(y|x)$ :

$$ p(y=1 | x) = \frac{1}{1 + e^{-xw}},$$$$p(y=0 | x) = 1 - p(y=1 | x) = \frac{1}{1 + e^{xw}}$$

\begin{align*} p(y=1|x) & = \frac{p(x|y=1)p(y=1)}{p(x)} \\ & = \frac{p(x|y=1)p(y=1)}{p(x|y=1)p(y=1)+p(x|y=0)p(y=0)} \\ & = \frac{1}{1+\frac{p(x|y=0)p(y=0)}{p(x|y=1)p(y=1)}} \\ & = \frac{1}{1+\frac{p(x|y=0)(1-p(y=1))}{p(x|y=1)p(y=1)}} \\ & = \frac{1}{1+\frac{p(x|y=0)(1-\pi)}{p(x|y=1)\pi}} \,\,\,\,\,\, (p(y=1)=\pi) \\ & = \frac{1}{1+e^{ln\frac{1-\pi}{\pi} + ln\frac{p(x|y=0)}{p(x|y=1)}}} \\ & = \frac{1}{1+e^{ln\frac{1-\pi}{\pi} + \sum_i ln\frac{p(x_i|y=0)}{p(x_i|y=1)}}} \end{align*}

$$ p(x_i|y_k) = \frac{1}{\sigma_{ik}\sqrt{2\pi}} e^{-\frac{{(x_i-\mu_{ik})}^2}{2{\sigma_{ik}}^2}}$$

\begin{align*} p(y=1|x) & = \frac{1}{1+e^{-(ln\frac{\pi-1}{\pi}+\sum_i(\frac{\mu_{i1}-\mu_{i0}}{\sigma^2_i}x_i + \frac{\mu^2_{i0}-\mu^2_{i1}}{2\sigma^2_i} ))}} \\ & = \frac{1}{1+e^{-\sum_iw_ix_i+b}}, \,\,\,\,\,\, i=1, 2, 3, , n\_features \end{align*}

$$ b = \sum_i(ln\frac{\pi-1}{\pi} + \frac{\mu^2_{i0}-\mu^2_{i1}}{2\sigma^2_i} )), \,\,\,\,\,\, w_i = \frac{\mu_{i1}-\mu_{i0}}{\sigma^2_i} $$

$$ w = agr \, min_w \sum_{i=1}^n y_i ln \, p(y_i=1|x_i) + (1 - y_i) ln \, p(y_i=0|x_i) $$

def shuffle_data(X, y, seed=None):
    if seed:
        np.random.seed(seed)

    idx = np.arange(X.shape[0])
    np.random.shuffle(idx)
    
    return X[idx], y[idx]



#  X
def normalize(X, axis=-1, p=2):
    lp_norm = np.atleast_1d(np.linalg.norm(X, p, axis))
    lp_norm[lp_norm == 0] = 1
    return X / np.expand_dims(lp_norm, axis)


#  X
def standardize(X):
    X_std = np.zeros(X.shape)
    mean = X.mean(axis=0)
    std = X.std(axis=0)
    
    # 0
    # X_std = (X - X.mean(axis=0)) / X.std(axis=0) 
    for col in range(np.shape(X)[1]):
        if std[col]:
            X_std[:, col] = (X_std[:, col] - mean[col]) / std[col]
    
    return X_std

def train_test_split(X, y, test_size=0.2, shuffle=True, seed=None):
    if shuffle:
        X, y = shuffle_data(X, y, seed)
        
    n_train_samples = int(X.shape[0] * (1-test_size))
    x_train, x_test = X[:n_train_samples], X[n_train_samples:]
    y_train, y_test = y[:n_train_samples], y[n_train_samples:]
    
    return x_train, x_test, y_train, y_test

def vec2diagonal(vec):
    vec_length = len(vec)
    diagonal = np.zeros((vec_length, vec_length))
    for i in range(vec_length):
        diagonal[i][i] = vec[i]
    return diagonal


def accuracy(y, y_pred):
    y = y.reshape(y.shape[0], -1)
    y_pred = y_pred.reshape(y_pred.shape[0], -1)
    return np.sum(y == y_pred)/len(y)


class Sigmoid:
    
    def function(self, x):
        return 1/(1 + np.exp(-x))
    
    def derivative(self, x):
        return self.function(x) * (1 - self.function(x))


class LogisticRegression():
    """. 
    Parameters:
    -----------
    learning_rate: float
        .
    gradient_descent: boolean
        wgradient_descent=True, gradient_descentw; 
        w.
    """
    def __init__(self, learning_rate=.1, gradient_descent=True):
        self.w = None
        self.learning_rate = learning_rate
        self.gradient_descent = gradient_descent
        self.sigmoid = Sigmoid()


    def fit(self, X, y, n_iterations=4000):
        # 1
        X = np.insert(X, 0, 1, axis=1)
        X = X.reshape(X.shape[0], -1)
        y = y.reshape(y.shape[0], -1)

        n_samples, n_features = np.shape(X)

        #  [-1/n_features, 1/n_features]
        limit = 1 / math.sqrt(n_features)
        self.w = np.random.uniform(-limit, limit, (n_features, 1))
        
        
        for i in range(n_iterations):
            # w
            y_pred = self.sigmoid.function(X.dot(self.w))
            # w.
            if self.gradient_descent:
                self.w -= self.learning_rate * X.T.dot(-(y - y_pred))
                self.w -= self.learning_rate * X.T.dot(-(y - y_pred) * 
                                 self.sigmoid.function(X.dot(self.w)) * 
                                (1 - self.sigmoid.function(X.dot(self.w))))
            # 
            else:
                # Make a diagonal matrix of the sigmoid gradient column vector
                diag_gradient = vec2diagonal(self.sigmoid.derivative(X.dot(self.w)))
                # Batch opt:
                self.w = np.linalg.pinv(X.T.dot(diag_gradient).dot(X)).dot(X.T).dot(diag_gradient.dot(X).dot(self.w) + y - y_pred)
                
    def predict(self, X):
        # 
        X = X.reshape(X.shape[0], -1)
        X = np.insert(X, 0, 1, axis=1)
        # 
        y_pred = np.round(self.sigmoid.function(X.dot(self.w))).astype(int)
        return y_pred
        

def main():
    # Load dataset
    data = datasets.load_iris()
    X = normalize(data.data[data.target != 0])
    y = data.target[data.target != 0]

    X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.33, seed=1)

if __name__ == "__main__":
